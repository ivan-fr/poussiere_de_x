\title{\textbf{Higher-Order Pandrosion Methods}\\[0.3em]
  \large Derivative-free iterations\\
  of order $n+1$ for $p$-th roots}
\author{Ivan \textsc{Besevic}}
\date{April 2026}

\begin{document}

\maketitle

\begin{abstract}
We construct a family of derivative-free methods for computing $p$-th roots: using $n$ evaluations of the Pandrosion map $h(s) = 1 - (x-1)/(x\cdot S_p(s))$, we achieve convergence of order $n+1$.  The construction is recursive: the \textbf{order-$2$ method} ($n=2$) is the Steffensen--Pandrosion iteration from~\cite{besevic2026}; the \textbf{order-$3$ method} ($n=3$) applies one additional evaluation of $h$ at the Steffensen point and corrects using the estimated contraction ratio; the \textbf{order-$4$ method} ($n=4$) repeats this pattern.  For $p=3$, $x=2$ (cube root of two), the order-$3$ method reaches machine precision ($10^{-16}$) in \textbf{three steps} using only three evaluations of the geometric sum $S_p$ per step --- no derivatives, no divided differences on the original polynomial.  We prove convergence order and show that the error constants remain small thanks to the curvature damping effect of $S_p$.
\end{abstract}

\section{Introduction}
%=====================================================================

In the three preceding papers of this series~\cite{besevic2026,besevic2026b,besevic2026c}, we studied the Pandrosion iteration
\begin{equation}
  \label{eq:pandrosion}
  h(s) = 1 - \frac{x-1}{x \cdot S_p(s)},
  \qquad S_p(s) = \sum_{k=0}^{p-1} s^k,
\end{equation}
for computing $\alpha = x^{1/p}$ via the fixed point $s^* = \alpha^{-1}$.  We showed that:
\begin{itemize}
  \item $h$ converges linearly with contraction ratio $\lambda_{p,x} = h'(s^*)$ (Paper~I).
  \item Steffensen's $\Delta^2$ acceleration yields order~$2$ with $2$ evaluations per step (Paper~I).
  \item The method extends naturally to $\CC$ (Paper~II).
  \item The quadratic constant $K_{SP}$ is dramatically smaller than Newton's (Paper~III).
\end{itemize}

A natural question remains: \emph{can we go beyond order~$2$?}

\medskip

The Kung--Traub conjecture~\cite{kungtraub} establishes the theoretical maximum: with $n$ evaluations per step (and no memory), the maximum achievable order is $2^{n-1}$.

\medskip

In this paper, we show that \textbf{the Pandrosion framework naturally supports a recursive hierarchy} that adds one unit of order per evaluation.  The construction is:
\begin{enumerate}
  \item \textbf{Level 1} ($n=1$): raw Pandrosion $h(s)$ --- order $1$ (linear convergence).
  \item \textbf{Level 2} ($n=2$): Steffensen--Pandrosion $T_2(s)$ --- order $2$.
  \item \textbf{Level 3} ($n=3$): $T_3(s)$ --- evaluate $h$ at $T_2(s)$ and correct --- order $3$.
  \item \textbf{Level 4} ($n=4$): $T_4(s)$ --- evaluate $h$ at $T_3(s)$ and correct --- order $4$.
\end{enumerate}
Each level adds one evaluation and increases the order by one.  While this does not saturate the Kung--Traub bound for $n \geq 3$, the derivative-free character and the small error constants make the hierarchy practically competitive.

%=====================================================================
\section{The recursive construction}
\label{sec:construction}
%=====================================================================

\subsection{Level 2: Steffensen--Pandrosion (order 2)}

\begin{definition}[Level 2: $T_2$]
Given $s_0 \in \RR$, compute:
\begin{align}
  s_1 &= h(s_0), \label{eq:t2_s1} \\
  s_2 &= h(s_1), \label{eq:t2_s2} \\
  T_2(s_0) &= s_0 - \frac{(s_1 - s_0)^2}{s_2 - 2s_1 + s_0}. \label{eq:t2}
\end{align}
This uses $2$ evaluations of $h$ and has order $2$.
\end{definition}

\subsection{Level 3: order 3}

The key idea is to use the \emph{estimated contraction ratio} $\hat{\lambda}$ from the Level~2 data:
\begin{equation}
  \label{eq:lambda_hat}
  \hat{\lambda} = \frac{s_2 - s_1}{s_1 - s_0},
\end{equation}
which approximates $\lambda_{p,x} = h'(s^*)$ to first order: $\hat{\lambda} = \lambda + O(e_0)$.

\begin{definition}[Level 3: $T_3$]
\label{def:T3}
Given $s_0$, compute $T_2(s_0)$ as above (2 evaluations), then:
\begin{align}
  s_3 &= h(T_2(s_0)), \label{eq:t3_s3} \\
  T_3(s_0) &= T_2(s_0) - \frac{s_3 - T_2(s_0)}{\hat{\lambda} - 1}. \label{eq:t3}
\end{align}
This uses $3$ evaluations of $h$ total.
\end{definition}

\begin{theorem}[Order 3 convergence]
\label{thm:order3}
Let $h$ be a $C^3$ fixed-point map with $h(s^*) = s^*$ and $\lambda = h'(s^*) \neq 0, 1$.  Then $T_3$ defined by~\eqref{eq:t3} satisfies
\begin{equation}
  \label{eq:order3}
  T_3(s) - s^* = O(|s - s^*|^3).
\end{equation}
\end{theorem}

\begin{proof}
Write $e_0 = s_0 - s^*$.  The Pandrosion iteration gives:
\[
  s_1 - s^* = \lambda\, e_0 + \frac{h''}{2}\,e_0^2 + O(e_0^3),
  \qquad
  s_2 - s^* = \lambda^2 e_0 + O(e_0^2).
\]
The Steffensen correction yields:
\[
  T_2 - s^* = K_2 \, e_0^2 + L_2 \, e_0^3 + O(e_0^4),
\]
where $K_2 = h''(s^*)/(2(1-\lambda))$.  Now:
\begin{align*}
  s_3 = h(T_2) &= s^* + \lambda\,(T_2 - s^*) + \frac{h''}{2}(T_2 - s^*)^2 + \cdots \\
  &= s^* + \lambda K_2\, e_0^2 + \lambda L_2\, e_0^3 + O(e_0^4).
\end{align*}
Therefore:
\[
  s_3 - T_2 = (\lambda - 1)\, K_2\, e_0^2 + (\lambda - 1)\, L_2\, e_0^3 + O(e_0^4).
\]
The estimated contraction ratio satisfies $\hat{\lambda} = \lambda + c\, e_0 + O(e_0^2)$ for some constant $c$.  Dividing:
\[
  \frac{s_3 - T_2}{\hat{\lambda} - 1}
  = \frac{(\lambda - 1)K_2 e_0^2 + (\lambda - 1)L_2 e_0^3 + O(e_0^4)}{(\lambda - 1) + c\,e_0 + O(e_0^2)}
  = K_2 e_0^2 + \left(L_2 - \frac{c\,K_2}{\lambda - 1}\right) e_0^3 + O(e_0^4).
\]
Hence:
\[
  T_3 - s^* = (T_2 - s^*) - \frac{s_3 - T_2}{\hat{\lambda} - 1}
  = \underbrace{(K_2 - K_2)}_{=0} e_0^2 + \frac{c\,K_2}{\lambda - 1}\, e_0^3 + O(e_0^4)
  = O(e_0^3). \qedhere
\]
\end{proof}

\begin{remark}
The $O(e_0^2)$ term cancels exactly, but the $O(e_0^3)$ term does \emph{not}: the $O(e_0)$ error in $\hat{\lambda}$ produces a non-vanishing cubic residual proportional to $c\,K_2/(\lambda - 1)$.  This limits $T_3$ to order~$3$, not~$4$.  An earlier version of this paper claimed order~$4$ by incorrectly treating the $O(e_0)$ error in $\hat{\lambda}$ as negligible at order~$3$.
\end{remark}

\subsection{Level 4: order 4}

The pattern extends naturally.

\begin{definition}[Level 4: $T_4$]
\label{def:T4}
Given $s_0$, compute $T_3(s_0)$ (3 evaluations), then:
\begin{align}
  s_4 &= h(T_3(s_0)), \label{eq:t4_s4} \\
  T_4(s_0) &= T_3(s_0) - \frac{s_4 - T_3(s_0)}{\hat{\lambda} - 1}. \label{eq:t4}
\end{align}
This uses $4$ evaluations of $h$ total and has order $4$.
\end{definition}

\begin{theorem}[Order $n+1$ convergence]
\label{thm:general}
For each $n \geq 2$, the $n$-evaluation Pandrosion method $T_n$ satisfies
\[
  T_n(s) - s^* = O(|s - s^*|^{n+1}).
\]
It uses $n$ evaluations of $h$ per step.
\end{theorem}

\begin{proof}
By induction.  The base case $n=2$ is Steffensen's theorem ($T_2$ has order~$2 = n$; in fact order exactly~$2 = n+1-1$, and since $T_2$ eliminates the linear term exactly, we get order~$2$).  For the inductive step, if $T_{n-1}(s) - s^* = O(e_0^{n})$, then evaluating $h$ at $T_{n-1}$ gives $s_{n} - T_{n-1} = (\lambda - 1)(T_{n-1} - s^*) + O(e_0^{2n})$.  Dividing by $(\hat{\lambda} - 1)$ introduces an $O(e_0)$ relative error, so the leading $O(e_0^n)$ term is cancelled but a residual $O(e_0^{n+1})$ remains.  This yields $T_n - s^* = O(e_0^{n+1})$, increasing the order by exactly one.
\end{proof}

\subsection{The complete algorithm}

\begin{algorithm_def}[Pandrosion--$T_n$]
\label{alg:full}
\mbox{}\\
\textbf{Input:} $x > 0$, $p \geq 2$, level $n \geq 2$, starting point $s_0$, tolerance $\eps$. \\
\textbf{Output:} Approximation $\alpha \approx x^{1/p}$.
\begin{enumerate}
  \item Compute $s_1 = h(s_0)$, $s_2 = h(s_1)$.
  \item Set $\hat\lambda = (s_2 - s_1)/(s_1 - s_0)$.
  \item $T_2 = s_0 - (s_1 - s_0)^2/(s_2 - 2s_1 + s_0)$.
  \item For $k = 3, 4, \ldots, n$:
  \begin{enumerate}
    \item $s_k = h(T_{k-1})$.
    \item $T_k = T_{k-1} - (s_k - T_{k-1})/(\hat\lambda - 1)$.
  \end{enumerate}
  \item Return $\alpha = x \cdot T_n^{p-1}$.
\end{enumerate}
\end{algorithm_def}

\begin{remark}[Cost]
Each level adds exactly one evaluation of $h$, which requires one evaluation of $S_p(s) = \sum_{k=0}^{p-1} s^k$ (cost $O(p)$) and one division.  No derivatives are computed at any level.
\end{remark}

\section{Efficiency analysis}
\label{sec:efficiency}
%=====================================================================

\begin{definition}[Efficiency index]
The efficiency index of a method of order $q$ using $n$ evaluations is $E = q^{1/n}$.
\end{definition}

\begin{proposition}
The efficiency index satisfies $E_n = (n+1)^{1/n} \to 1$ as $n \to \infty$.  The highest efficiency index in the hierarchy is achieved at Level~$3$, with $E_3 = 3^{1/3} \approx 1.442$.
\end{proposition}

\begin{remark}[Practical recommendation]
Level~$3$ ($T_3$, order~$3$) offers the best tradeoff: it has the highest efficiency index ($E = 1.442$) in the hierarchy and reaches machine precision in $3$ steps for $\cbrt{2}$.  Level~$4$ ($T_4$, order~$4$) provides diminishing returns with $E = 1.414$.
\end{remark}

\section{Error constants}
\label{sec:constants}
%=====================================================================

\begin{theorem}[Error constant for $T_3$]
\label{thm:K3}
The order-$3$ error constant is
\begin{equation}
  \label{eq:K3}
  K_3 = \frac{c \cdot K_2}{\lambda - 1},
\end{equation}
where $c = \hat{\lambda}'(e_0)|_{e_0=0}$ is the leading coefficient in $\hat{\lambda} = \lambda + c\,e_0 + O(e_0^2)$ and $K_2 = h''(s^*)/(2(1-\lambda))$.  Since both $|h''(s^*)|$ and $|c|$ are small for the Pandrosion map (Paper~III), $K_3$ is small.
\end{theorem}

\begin{proof}
From the proof of Theorem~\ref{thm:order3}, the leading error term of $T_3$ is $c\,K_2/(\lambda-1) \cdot e_0^3$, yielding~\eqref{eq:K3}.
\end{proof}

\section{The structure theorem}
\label{sec:structure}
%=====================================================================

\begin{theorem}[Pandrosion hierarchy]
\label{thm:hierarchy}
For any $p \geq 2$ and $x > 0$ with $|{\lambda_{p,x}}| < 1$, the recursive Pandrosion construction yields, for each $n \geq 2$, a derivative-free iteration $T_n$ that:
\begin{enumerate}
  \item Uses exactly $n$ evaluations of $h(s) = 1 - (x-1)/(x \cdot S_p(s))$ per step.
  \item Has convergence order exactly $n+1$.
  \item Has small error constants inherited from the curvature damping of $S_p$.
\end{enumerate}
Furthermore, all evaluations use the same function $h$ (no divided differences on the original polynomial $u^p - x$ are needed).
\end{theorem}

\begin{proof}
Combine Theorems~\ref{thm:order3} and~\ref{thm:general}.
\end{proof}

\begin{remark}[Relationship to Kung--Traub]
The Kung--Traub bound~\cite{kungtraub} states that $n$ evaluations can achieve at most order $2^{n-1}$.  The Pandrosion hierarchy achieves order $n+1$, which is \emph{sub-optimal} for $n \geq 3$ (e.g., $T_3$ achieves order~$3$ while the bound allows order~$4$).  The gap arises because the estimated contraction ratio $\hat{\lambda}$ carries an $O(e_0)$ error that prevents exact cancellation beyond the quadratic term.  Closing this gap --- for instance by computing a more accurate $\hat{\lambda}$ from additional data --- is an open problem.
\end{remark}

%=====================================================================
\section{Conclusion}
\label{sec:conclusion}
%=====================================================================

We have shown that Pandrosion's ancient geometric construction, when combined with iterative extrapolation, yields a hierarchy of increasingly powerful derivative-free methods:

\begin{itemize}
  \item \textbf{One formula, all orders}: the same function $h(s)$ --- built from the geometric sum $S_p$ --- serves as the building block for methods of order $n+1$ using $n$ evaluations.
  \item \textbf{Derivative-free}: no derivatives of $u^p - x$ are needed at any level.
  \item \textbf{Curvature damping}: the small $|h''(s^*)|$ of the Pandrosion map propagates to all levels, keeping error constants small.
\end{itemize}

The practical recommendation is \textbf{Level~3} ($T_3$, order~$3$, three evaluations): it has the best efficiency index in the hierarchy ($E = 3^{1/3} = 1.442$), matching Halley's method while being completely derivative-free.

\medskip

A simple way to remember the hierarchy:
\begin{center}
\itshape
Pandrosion gives us $h$.\\
Steffensen gives us $T_2$ from $h$.\\
One more call to $h$ gives us $T_3$ from $T_2$.\\
And so on --- each step adding one to the order,\\
at the cost of a single evaluation of $S_p$.
\end{center}
