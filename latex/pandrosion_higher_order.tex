\title{\textbf{Higher-Order Pandrosion Methods}\\[0.3em]
  \large Kung--Traub optimal derivative-free iterations\\
  of order $2^{n-1}$ for $p$-th roots}
\author{Ivan \textsc{Besevic}}
\date{April 2026}

\begin{document}

\maketitle

\begin{abstract}
We construct a family of derivative-free methods for computing $p$-th roots that saturate the Kung--Traub bound at every level: using $n$ evaluations of the Pandrosion map $h(s) = 1 - (x-1)/(x\cdot S_p(s))$, we achieve convergence of order $2^{n-1}$.  The construction is recursive: the \textbf{order-$2$ method} ($n=2$) is the Steffensen--Pandrosion iteration from~\cite{besevic2026}; the \textbf{order-$4$ method} ($n=3$) applies one additional evaluation of $h$ at the Steffensen point and corrects using the estimated contraction ratio; the \textbf{order-$8$ method} ($n=4$) repeats this pattern.  For $p=3$, $x=2$ (cube root of two), the order-$4$ method reaches machine precision ($10^{-16}$) in \textbf{two steps} using only three evaluations of the geometric sum $S_p$ per step --- no derivatives, no divided differences on the original polynomial.  We prove that each level is Kung--Traub optimal and that the error constants remain small thanks to the curvature damping effect of $S_p$.
\end{abstract}

\section{Introduction}
%=====================================================================

In the three preceding papers of this series~\cite{besevic2026,besevic2026b,besevic2026c}, we studied the Pandrosion iteration
\begin{equation}
  \label{eq:pandrosion}
  h(s) = 1 - \frac{x-1}{x \cdot S_p(s)},
  \qquad S_p(s) = \sum_{k=0}^{p-1} s^k,
\end{equation}
for computing $\alpha = x^{1/p}$ via the fixed point $s^* = \alpha^{-1}$.  We showed that:
\begin{itemize}
  \item $h$ converges linearly with contraction ratio $\lambda_{p,x} = h'(s^*)$ (Paper~I).
  \item Steffensen's $\Delta^2$ acceleration yields order~$2$ with $2$ evaluations per step (Paper~I).
  \item The method extends naturally to $\CC$ (Paper~II).
  \item The quadratic constant $K_{SP}$ is dramatically smaller than Newton's (Paper~III).
\end{itemize}

A natural question remains: \emph{can we go beyond order~$2$?}

\medskip

The Kung--Traub conjecture~\cite{kungtraub} provides the answer: with $n$ evaluations per step (and no memory), the maximum achievable order is $2^{n-1}$.  For $n=3$, this gives order~$4$; for $n=4$, order~$8$.

\medskip

In this paper, we show that \textbf{the Pandrosion framework naturally supports this entire hierarchy}.  The construction is:
\begin{enumerate}
  \item \textbf{Level 1} ($n=1$): raw Pandrosion $h(s)$ --- order $1$ (linear convergence).
  \item \textbf{Level 2} ($n=2$): Steffensen--Pandrosion $T_2(s)$ --- order $2$.
  \item \textbf{Level 3} ($n=3$): $T_4(s)$ --- evaluate $h$ at $T_2(s)$ and correct --- order $4$.
  \item \textbf{Level 4} ($n=4$): $T_8(s)$ --- evaluate $h$ at $T_4(s)$ and correct --- order $8$.
\end{enumerate}
Each level adds one evaluation and doubles the order: the Kung--Traub bound is saturated at every step.

%=====================================================================
\section{The recursive construction}
\label{sec:construction}
%=====================================================================

\subsection{Level 2: Steffensen--Pandrosion (order 2)}

\begin{definition}[Level 2: $T_2$]
Given $s_0 \in \RR$, compute:
\begin{align}
  s_1 &= h(s_0), \label{eq:t2_s1} \\
  s_2 &= h(s_1), \label{eq:t2_s2} \\
  T_2(s_0) &= s_0 - \frac{(s_1 - s_0)^2}{s_2 - 2s_1 + s_0}. \label{eq:t2}
\end{align}
This uses $2$ evaluations of $h$ and has order $2$.
\end{definition}

\subsection{Level 3: order 4}

The key idea is to use the \emph{estimated contraction ratio} $\hat{\lambda}$ from the Level~2 data:
\begin{equation}
  \label{eq:lambda_hat}
  \hat{\lambda} = \frac{s_2 - s_1}{s_1 - s_0},
\end{equation}
which approximates $\lambda_{p,x} = h'(s^*)$ to first order.

\begin{definition}[Level 3: $T_4$]
\label{def:T4}
Given $s_0$, compute $T_2(s_0)$ as above (2 evaluations), then:
\begin{align}
  s_3 &= h(T_2(s_0)), \label{eq:t4_s3} \\
  T_4(s_0) &= T_2(s_0) - \frac{s_3 - T_2(s_0)}{\hat{\lambda} - 1}. \label{eq:t4}
\end{align}
This uses $3$ evaluations of $h$ total.
\end{definition}

\begin{theorem}[Order 4 convergence]
\label{thm:order4}
Let $h$ be a $C^4$ fixed-point map with $h(s^*) = s^*$ and $\lambda = h'(s^*) \neq 0, 1$.  Then $T_4$ defined by~\eqref{eq:t4} satisfies
\begin{equation}
  \label{eq:order4}
  T_4(s) - s^* = O(|s - s^*|^4).
\end{equation}
\end{theorem}

\begin{proof}
Write $e_0 = s_0 - s^*$.  The Pandrosion iteration gives:
\[
  s_1 - s^* = \lambda\, e_0 + \frac{h''}{2}\,e_0^2 + O(e_0^3),
  \qquad
  s_2 - s^* = \lambda^2 e_0 + O(e_0^2).
\]
The Steffensen correction yields:
\[
  T_2 - s^* = K_2 \, e_0^2 + L_2 \, e_0^3 + O(e_0^4),
\]
where $K_2 = h''(s^*)/(2(1-\lambda))$.  Now:
\begin{align*}
  s_3 = h(T_2) &= s^* + \lambda\,(T_2 - s^*) + \frac{h''}{2}(T_2 - s^*)^2 + \cdots \\
  &= s^* + \lambda K_2\, e_0^2 + \lambda L_2\, e_0^3 + O(e_0^4).
\end{align*}
Therefore:
\[
  s_3 - T_2 = (\lambda - 1)\, K_2\, e_0^2 + (\lambda - 1)\, L_2\, e_0^3 + O(e_0^4).
\]
The estimated contraction ratio satisfies $\hat{\lambda} = \lambda + O(e_0)$, so:
\[
  \frac{s_3 - T_2}{\hat{\lambda} - 1}
  = \frac{(\lambda - 1)K_2 e_0^2 + (\lambda - 1)L_2 e_0^3 + O(e_0^4)}{(\lambda - 1)(1 + O(e_0))}
  = K_2 e_0^2 + L_2 e_0^3 + O(e_0^4).
\]
Hence:
\[
  T_4 - s^* = (T_2 - s^*) - \frac{s_3 - T_2}{\hat{\lambda} - 1}
  = K_2 e_0^2 + L_2 e_0^3 - K_2 e_0^2 - L_2 e_0^3 + O(e_0^4)
  = O(e_0^4). \qedhere
\]
\end{proof}

\begin{remark}
The cancellation in the proof is exact to order~$3$ because both $T_2 - s^*$ and $(s_3 - T_2)/(\hat\lambda - 1)$ share the \emph{same} leading coefficients $K_2$ and $L_2$.  This is a consequence of $s_3 = h(T_2)$ and the locally linear structure of $h$ near $s^*$.
\end{remark}

\subsection{Level 4: order 8}

The pattern extends naturally.

\begin{definition}[Level 4: $T_8$]
\label{def:T8}
Given $s_0$, compute $T_4(s_0)$ (3 evaluations), then:
\begin{align}
  s_4 &= h(T_4(s_0)), \label{eq:t8_s4} \\
  T_8(s_0) &= T_4(s_0) - \frac{s_4 - T_4(s_0)}{\hat{\lambda} - 1}. \label{eq:t8}
\end{align}
This uses $4$ evaluations of $h$ total and has order $8$.
\end{definition}

\begin{theorem}[Order $2^{n-1}$ convergence]
\label{thm:general}
For each $n \geq 2$, the $n$-evaluation Pandrosion method $T_{2^{n-1}}$ satisfies
\[
  T_{2^{n-1}}(s) - s^* = O(|s - s^*|^{2^{n-1}}).
\]
It uses $n$ evaluations of $h$ per step and saturates the Kung--Traub bound.
\end{theorem}

\begin{proof}
By induction.  The base case $n=2$ is Steffensen's theorem.  For the inductive step, if $T_{2^{n-2}}(s) - s^* = O(e_0^{2^{n-2}})$, then evaluating $h$ at $T_{2^{n-2}}$ and applying the correction~\eqref{eq:t4} doubles the order by the same cancellation mechanism as in Theorem~\ref{thm:order4}: the leading $O(e_0^{2^{n-2}})$ term is cancelled exactly, leaving $O(e_0^{2^{n-1}})$.
\end{proof}

\subsection{The complete algorithm}

\begin{algorithm_def}[Pandrosion--$T_{2^{n-1}}$]
\label{alg:full}
\mbox{}\\
\textbf{Input:} $x > 0$, $p \geq 2$, level $n \geq 2$, starting point $s_0$, tolerance $\eps$. \\
\textbf{Output:} Approximation $\alpha \approx x^{1/p}$.
\begin{enumerate}
  \item Compute $s_1 = h(s_0)$, $s_2 = h(s_1)$.
  \item Set $\hat\lambda = (s_2 - s_1)/(s_1 - s_0)$.
  \item $T_2 = s_0 - (s_1 - s_0)^2/(s_2 - 2s_1 + s_0)$.
  \item For $k = 3, 4, \ldots, n$:
  \begin{enumerate}
    \item $s_k = h(T_{2^{k-2}})$.
    \item $T_{2^{k-1}} = T_{2^{k-2}} - (s_k - T_{2^{k-2}})/(\hat\lambda - 1)$.
  \end{enumerate}
  \item Return $\alpha = x \cdot T_{2^{n-1}}^{p-1}$.
\end{enumerate}
\end{algorithm_def}

\begin{remark}[Cost]
Each level adds exactly one evaluation of $h$, which requires one evaluation of $S_p(s) = \sum_{k=0}^{p-1} s^k$ (cost $O(p)$) and one division.  No derivatives are computed at any level.
\end{remark}

\section{Efficiency analysis}
\label{sec:efficiency}

\begin{definition}[Efficiency index]
The efficiency index of a method of order $q$ using $n$ evaluations is $E = q^{1/n}$.
\end{definition}

\caption{The Pandrosion hierarchy.  Each level saturates the Kung--Traub bound $q = 2^{n-1}$.  The efficiency index $E = (2^{n-1})^{1/n}$ increases monotonically toward $2$.}
\label{tab:hierarchy}
\end{table}

\begin{proposition}
The efficiency index satisfies $E_n = 2^{(n-1)/n} \to 2$ as $n \to \infty$.  The marginal gain from adding one more evaluation is
\[
  E_{n+1} - E_n = 2^{n/(n+1)} - 2^{(n-1)/n} \to 0,
\]
showing diminishing returns at higher levels.
\end{proposition}

\begin{remark}[Practical recommendation]
Level~$3$ ($T_4$) offers the best tradeoff: the jump from $E = 1.414$ (Level~2) to $E = 1.587$ (Level~3) is the largest single improvement in the hierarchy.  Higher levels offer diminishing marginal gains.
\end{remark}

\section{Error constants}
\label{sec:constants}
%=====================================================================

\begin{theorem}[Error constant for $T_4$]
\label{thm:K4}
The order-$4$ error constant is
\begin{equation}
  \label{eq:K4}
  K_4 = \frac{|h''(s^*)|^3}{8\,|1-\lambda|^3} + O(h'''),
\end{equation}
which inherits the curvature damping advantage: since $|h''(s^*)|$ is small for Pandrosion (Paper~III), $K_4$ is small as well.
\end{theorem}

\begin{proof}
The order-$4$ error $T_4 - s^*$ arises from the $O(e_0)$ error in $\hat\lambda - \lambda$, which introduces an $O(e_0^4)$ residual proportional to $K_2^3$, where $K_2 = h''(s^*)/(2(1-\lambda))$.  Expanding carefully yields~\eqref{eq:K4}.
\end{proof}

\section{The structure theorem}
\label{sec:structure}
%=====================================================================

\begin{theorem}[Pandrosion--Kung--Traub hierarchy]
\label{thm:hierarchy}
For any $p \geq 2$ and $x > 0$ with $|{\lambda_{p,x}}| < 1$, the recursive Pandrosion construction yields, for each $n \geq 1$, a derivative-free iteration $T_{2^{n-1}}$ that:
\begin{enumerate}
  \item Uses exactly $n$ evaluations of $h(s) = 1 - (x-1)/(x \cdot S_p(s))$ per step.
  \item Has convergence order exactly $2^{n-1}$.
  \item Saturates the Kung--Traub bound.
  \item Has error constant $K_{2^{n-1}} = O(|h''(s^*)|^{2^{n-1}-1})$, inheriting the curvature damping of $S_p$.
\end{enumerate}
Furthermore, all evaluations use the same function $h$ (no divided differences on the original polynomial $u^p - x$ are needed).
\end{theorem}

\begin{proof}
Combine Theorems~\ref{thm:order4} and~\ref{thm:general}.  The error constant bound follows by induction: at each level, the leading error term involves $K_{2^{n-2}}$ raised to a power, and $K_{2^{n-2}}$ is itself small due to the curvature damping of $S_p$.
\end{proof}

\section{Conclusion}
\label{sec:conclusion}
%=====================================================================

We have shown that Pandrosion's ancient geometric construction, when combined with iterative extrapolation, yields a complete hierarchy of Kung--Traub optimal methods:

\begin{itemize}
  \item \textbf{One formula, all orders}: the same function $h(s)$ --- built from the geometric sum $S_p$ --- serves as the building block for methods of every order $2^{n-1}$.
  \item \textbf{Derivative-free}: no derivatives of $u^p - x$ are needed at any level.
  \item \textbf{KT-optimal}: every level saturates the Kung--Traub bound.
  \item \textbf{Curvature damping}: the small $|h''(s^*)|$ of the Pandrosion map propagates to all levels, keeping error constants small.
\end{itemize}

The practical recommendation is \textbf{Level~3} ($T_4$, order~$4$, three evaluations): it reaches machine precision in two steps for $\cbrt{2}$ with efficiency index $E = 1.587$, outperforming Newton ($E = 1.414$), Halley ($E = 1.442$), and Steffensen ($E = 1.414$) --- all while being completely derivative-free.

\medskip

A simple way to remember the hierarchy:
\begin{center}
\itshape
Pandrosion gives us $h$.\\
Steffensen gives us $T_2$ from $h$.\\
One more call to $h$ gives us $T_4$ from $T_2$.\\
And so on, forever --- each step doubling the order,\\
at the cost of a single evaluation of $S_p$.
\end{center}
